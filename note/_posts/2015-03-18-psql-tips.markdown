#### List all databases:
{% highlight bash %}
$ \list
$ \l
{% endhighlight %}

#### List all tables in the current database:
{% highlight bash %}
$ \dt
{% endhighlight %}

#### Switch databases:
{% highlight bash %}
$ \connect database_name
{% endhighlight %}
